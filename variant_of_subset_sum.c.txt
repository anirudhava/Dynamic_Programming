counting the number of pairs of subsets whose difference of sum is the given value

sum(s2)+sum(s1)=sum(arr)
sum(s2)-sum(s1)=diff
---------------------------adding
sum(s2)=.5(sum(arr)+diff)

the problem boils down to finding all the subets of the array whose sum is =.5(sum(arr)+diff)

int count_pair(int* arr,int n,int diff)
{
  int sum=0;
	
  for(int i=0;i<n;i++)
  {
	sum+=arr[i];
  }

  return subset_sum_variant(arr,n,(sum+diff)/2);	
}

int subset_sum_variant(int* arr,int n,int sum)//cal the number of subsets with SUM=sum
{
  int dp[n+1][sum+1];

  for(int j=0;j<=sum;j++)
  {
	dp[0][j]=0;
  }

  for(int i=0;i<=n;i++)
  {
 	dp[i][0]=1;
  }

  for(int i=1;i<=n;i++)
  {
    for(int j=1;j<=sum;j++)
    {
	if(arr[i-1]<=j)
	{
		dp[i][j]=dp[i-1][j]+dp[i-1][j-arr[i-1]];	
	}
	else//arr[i-1]>j
	{
		dp[i][j]=dp[i-1][j];
	}
    }
  }

  return dp[n][sum];
}
